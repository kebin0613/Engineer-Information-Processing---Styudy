    public class Gisa {
	public static void main(String[] args) {
		int arr[] = {15, 20, 35, 5, 10};
		int i, j, temp;
		for(i = 1; i < arr.length; i++) {
			for(j = 0; j < arr.length-i; j++) {
				if(arr[j] > arr[j+1]) {
					temp = arr[j];
					arr[j] = arr[j+1];
					arr[j+1] = temp;
				}
			}
		}
		for(i = 0; i < arr.length; i++) {
			System.out.println(arr[i]);
		}
	}
    }
